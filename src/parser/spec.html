Comment Parser Specification

<comment>             := <single-comment> (<single-comment>)*
<single-comment>      := description
                      | tag ('-' description | <parameter-list>)
                      | markdown

<parameter-list>      := formal-parameter | parameter-list, formal-parameter
<formal-parameter>    := <parameter> ('=' initializer) | <optional-parameter>

<parameter>           := identifier (':' type | ':' '(' type ')')
<optional-parameter>  := identifier '?' (':' type)
<type>                := any | union-type-list | intersect-type-list | arrow-function-type

<union-type>          := '|' type | union-type '|' type

<intersect-type>      := '&' type | intersect-type '&' type

<arrow-function-type> := '(' parameter-list ')' '=' '>' type
